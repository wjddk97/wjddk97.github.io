import itertools
    
def solution(nums):
    combinations = itertools.combinations(nums, 3)
        
        
    temporary = []
    for i in combinations:
        temporary.append(sum(i))
        
    count = len(temporary)
    for i in temporary:
        for j in range(2, i):
            if i % j == 0:
                count -= 1
                break
        
    return count
